	name = GetLeLaRelSec_CONVERTER
	text = {
		localisation_key = string_le_FR
		trigger = {
			frloc_secondary_religion_masc_cons = yes
		}
	}
	text = {
		localisation_key = string_la_FR
		trigger = {
			frloc_secondary_religion_fem_cons = yes
		}
	}
	text = {
		localisation_key = string_l_FR
		trigger = {
			OR = {
				frloc_secondary_religion_masc_vow = yes
				frloc_secondary_religion_fem_vow = yes
			}
		}
	}
	text = {
		localisation_key = string_le_FR
		trigger = {
			NOT = { frloc_secondary_religion_masc_cons = yes }
			NOT = { frloc_secondary_religion_fem_cons = yes }
		}
	}
}
