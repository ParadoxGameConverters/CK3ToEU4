country_event = {
	id = plati_flavor_ckeu.1

	title = "plati_flavor_ckeu.EVTNAME1"				#Christian Mission
	desc = "plati_flavor_ckeu.EVTDESC1"

	picture = RELIGION_eventPicture

	trigger = {
		NOT = {
			has_country_modifier = plati_first_event_timer				#Timer, so that event doesnt fire too often
		}
		AND = {
			religion_group = pagan
			capital_scope = {
				continent = europe
			}
		}
		any_owned_province = {
			is_state = yes
			is_capital = no
			has_owner_religion = yes
			NOT = { has_province_modifier = plati_christian_mission_modifier }
		}
		OR = {
			current_age = age_of_discovery
			current_age = age_of_reformation
		}
	}

	mean_time_to_happen = {
		months = 600
	}

	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = plati_first_event_timer
				duration = 3650
				hidden = yes
			}
		}
		random_owned_province = {
			limit = {
				is_state = yes
				is_capital = no
				has_owner_religion = yes
				NOT = { has_province_modifier = plati_christian_mission_modifier }
			}
			save_event_target_as = plati_mission_province
		}
	}

	option = {
		name = "plati_flavor_ckeu.EVTOPTA1"				#Very well, let them come
		ai_chance = {
			factor = 20
		}
		event_target:plati_mission_province = {
			add_permanent_province_modifier = {
				name = "plati_christian_mission_modifier"
				duration = -1
			}
			change_religion = catholic
			add_base_tax = 1
		}
	}
	option = {
		name = "plati_flavor_ckeu.EVTOPTB1"				#They are not welcome here
		ai_chance = {
			factor = 80
		}
		add_prestige = 5
	}
}
