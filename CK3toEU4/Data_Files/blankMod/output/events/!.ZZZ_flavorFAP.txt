# The Counter-Reformation and the Reformation of the Catholic Faith
country_event = {
	id = flavor_fap.7
	title = "flavor_pap.EVTNAME7"
	desc = "flavor_pap.EVTDESC7"
	picture = REFORM_eventPicture
	
	fire_only_once = yes
	
	major = yes
	# this = country seeing popup, from = original event scope
	major_trigger = {
		religion_group = christian
	}
	
	trigger = {
		tag = FAP
		
		OR = {
			POR = { has_country_modifier = the_societas_jesu }
			SPA = { has_country_modifier = the_societas_jesu }
			is_religion_enabled = reformed
		}
		is_religion_enabled = protestant
		OR = {
			AND = {
				is_religion_enabled = reformed
				religion_years = { reformed = 14 }
			}
			religion_years = { protestant = 30 }
		}
		any_known_country = {
			religion = protestant
		}
		any_known_country = {
			religion = fraticelli
		}
		if = {
			limit = {
				has_dlc = "Emperor"
			}
			is_council_enabled = no
		}
		NOT = { has_global_flag = counter_reformation }
	}
	
	mean_time_to_happen = {
		months = 100
	}
	
	option = {
		name = "flavor_pap.EVTOPTA7"
		add_prestige = 10
		add_years_of_income = -0.5
		if = {
			limit = {
				corruption = 0.01
			}
			add_corruption = -2
		}
		if = {
			limit = {
				has_dlc = "Emperor"
			}
			custom_tooltip = flavor_fap.7.tt
			hidden_effect = {
				every_known_country = {
					limit = {
						religion = fraticelli
						any_owned_province = {
							has_cardinal = yes
						}
					}
					country_event = {
						id = flavor_fap.8
					}
				}
			}
		}
		set_global_flag = counter_reformation
		if = {
			limit = {
				has_dlc = "Emperor"
			}
			enable_council = yes
		}
	}
}

# Sublimis Deus
# by Daniel Tolman
country_event = {
	id = flavor_fap.9
	title = "flavor_pap.EVTNAME9"
	desc = "flavor_pap.EVTDESC9"
	picture = POPE_PREACHING_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Golden Century"
		religion = fraticelli
		culture_group = iberian
		capital_scope = {
			continent = europe
		}
		calc_true_if = {
			all_subject_country = {
				is_colonial_nation = yes
			}
			amount = 3
		}
		any_subject_country = {
			is_colonial_nation = yes
			num_of_cities = 15
		}
		exists = FAP
	}
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					religion = fraticelli
					any_subject_country = {
						is_colonial_nation = yes
					}
				}
				country_event = { id = flavor_fap.10 days = 3 }
			}
		}
	}
	
	option = {
		name = "flavor_pap.9.a"
		add_papal_influence = 20
		FAP = {
			add_opinion = { who = ROOT modifier = opinion_sublimis_deus }
		}
		add_country_modifier = {
			name = "sublimis_deus"
			duration = 36500
		}
	}
	
	option = {
		name = "flavor_pap.9.b"
		every_subject_country = {
			limit = {
				is_colonial_nation = yes
			}
			add_country_modifier = {
				name = "sublimis_deus_not_enforced"
				duration = 36500
			}
		}
	}
}
